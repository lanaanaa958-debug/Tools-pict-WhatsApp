import android.app.Activity import android.content.Intent import android.net.Uri import android.os.Build import android.os.Bundle import android.os.Environment import android.provider.Settings import android.widget.Button import android.widget.Toast import androidx.appcompat.app.AppCompatActivity class MainActivity : AppCompatActivity() { private lateinit var btnEnableNotification: Button private lateinit var btnStartService: Button private lateinit var btnRequestStorage: Button companion object { private const val REQUEST_CODE_NOTIFICATION = 1001 private const val REQUEST_CODE_MANAGE_STORAGE = 1002 } override fun onCreate(savedInstanceState: Bundle?) { super.onCreate(savedInstanceState) setContentView(R.layout.activity_main) btnEnableNotification = findViewById(R.id.btnEnableNotification) btnStartService = findViewById(R.id.btnStartService) btnRequestStorage = findViewById(R.id.btnRequestStorage) btnEnableNotification.setOnClickListener { startActivity(Intent(Settings.ACTION_NOTIFICATION_LISTENER_SETTINGS)) } btnRequestStorage.setOnClickListener { if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R) { val intent = Intent(Settings.ACTION_MANAGE_APP_ALL_FILES_ACCESS_PERMISSION) intent.data = Uri.parse("package:$packageName") startActivityForResult(intent, REQUEST_CODE_MANAGE_STORAGE) } else { Toast.makeText(this, "Izin sudah diberikan secara otomatis di Android 10 ke bawah", Toast.LENGTH_LONG).show() } } btnStartService.setOnClickListener { checkPermissionsAndStart() } } private fun checkPermissionsAndStart() { // Cek izin MANAGE_EXTERNAL_STORAGE untuk Android 11+ if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R && !Environment.isExternalStorageManager()) { Toast.makeText(this, "Harap berikan izin 'Izinkan akses ke semua file' terlebih dahulu", Toast.LENGTH_LONG).show() return } // Cek apakah NotificationListener sudah diaktifkan if (!isNotificationListenerEnabled()) { Toast.makeText(this, "Aktifkan akses notifikasi melalui tombol di atas", Toast.LENGTH_LONG).show() return } // Mulai service val intent = Intent(this, RecoveryService::class.java) startService(intent) Toast.makeText(this, "Service pemulihan cache berjalan...", Toast.LENGTH_LONG).show() } private fun isNotificationListenerEnabled(): Boolean { val enabledListeners = Settings.Secure.getString(contentResolver, "enabled_notification_listeners") return enabledListeners?.contains(packageName) == true } override fun onActivityResult(requestCode: Int, resultCode: Int, data: Intent?) { super.onActivityResult(requestCode, resultCode, data) if (requestCode == REQUEST_CODE_MANAGE_STORAGE && resultCode == Activity.RESULT_OK) { if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.R && Environment.isExternalStorageManager()) { Toast.makeText(this, "Izin penyimpanan penuh diberikan", Toast.LENGTH_SHORT).show() } } } }
